LG1352 【没有上司的舞会】

Posted on 2021-04-17，2 min read

这题可以看做一颗树，因为不会有人上司的上司是自己。

用d[i]d[i]d[i]来描述第iii个人及他的下属的最高快乐指数和。

用a[i][j]a[i][j]a[i][j]来描述第iii个人的第jjj个下属。

每个人如果选上,他的下属就不能选上，要的就是他下属的下属的最优值，这样就没有直接上司了

如果不选上，他的下属就可以选上，要他下属的最优值

考虑继承快乐指数和最大的一种

代码如下:（不要抄袭）

#include<iostream>
#include<cmath>
using namespace std;
const int N=76801;
int h[6001],n,i,a[6001][5000],x,y,d[6001],m;
int dfs(int root,int s){
	if(s==0){
		if(d[root]==N)
			d[root]=max(dfs(root,1),dfs(root,2)+h[root]);
            //继承最好状态
		return d[root];
	}
	int l=0;
	for(int j=1;j<=a[root][0];j++)
		l+=dfs(a[root][j],s-1);
   //枚举他的下属
	return l;
}
int main(){
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>h[i];
		d[i]=N;
        //赋初值
	}	
	cin>>x>>y;
	while(x!=0||y!=0){
		a[y][++a[y][0]]=x;
		cin>>x>>y;
	}
	for(i=1;i<=n;i++){
		if(d[i]==N)
			d[i]=dfs(i,0);
        //枚举顶头上司
		m=max(m,d[i]);
	}
	cout<<m;
	return 0;	
}
